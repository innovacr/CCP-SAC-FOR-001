Cómo Conectar a tu Google Drive

  1. Ve a tu Google Drive y crea una nueva **Carpeta** o simplemente ve a tu cuenta de Google.
  2. Entra en .com y haz clic en "Nuevo proyecto".
  3. Sustituye todo el código por el siguiente bloque de **Google Apps Script** (que leerá y guardará tus datos en una hoja de Google Sheets automáticamente):
/*
 * SCRIPT PARA SPREADSHEET GOOGLE DRIVE - CONTROL RECLAMOS CCP
 * Copia y pega esto íntegramente en .com
 */
function doGet(e) {
  var sheet = getOrCreateSheet();
  var rows = sheet.getDataRange().getValues();
  var data = [];
  
  // Omitir fila de encabezado (index 0)
  for (var i = 1; i < rows.length; i++) {
    var row = rows[i];
    if (!row[0]) continue;
    data.push({
      consecutivo: row[0],
      fecha: row[1],
      filial: row[2].toString(),
      condicion: row[3],
      nombre: row[4],
      referencia: row[5],
      categoria: row[6],
      resumen: row[7],
      estatus: row[8],
      fechaCierre: row[9],
      seguimientos: JSON.parse(row[10] || "[]")
    });
  }
  
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var params = JSON.parse(e.postData.contents);
    var sheet = getOrCreateSheet();
    var rows = sheet.getDataRange().getValues();
    
    var targetConsecutivo = params.consecutivo;
    var rowNumber = -1;
    
    // Buscar si ya existe el consecutivo para actualizarlo
    for (var i = 1; i < rows.length; i++) {
      if (rows[i][0] === targetConsecutivo) {
        rowNumber = i + 1; // Apps Script es index 1
        break;
      }
    }
    
    var rowData = [
      params.consecutivo,
      params.fecha,
      params.filial,
      params.condicion,
      params.nombre,
      params.referencia,
      params.categoria,
      params.resumen,
      params.estatus,
      params.fechaCierre,
      JSON.stringify(params.seguimientos || [])
    ];
    
    if (rowNumber !== -1) {
      // Actualizar fila existente
      sheet.getRange(rowNumber, 1, 1, rowData.length).setValues([rowData]);
    } else {
      // Insertar nuevo reclamo
      sheet.appendRow(rowData);
    }
    
    return ContentService.createTextOutput(JSON.stringify({ status: "success" }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({ status: "error", message: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function getOrCreateSheet() {
  var docName = "Control_Reclamos_CCP_DB";
  var files = DriveApp.getFilesByName(docName);
  var ss;
  if (files.hasNext()) {
    ss = SpreadsheetApp.open(files.next());
  } else {
    ss = SpreadsheetApp.create(docName);
    var sheet = ss.getActiveSheet();
    sheet.appendRow([
      "Consecutivo", "Fecha Rec.", "N° Filial", "Condición", 
      "Nombre Reclamante", "Referencia", "Categoría", "Resumen del Reclamo",
      "Estatus Actual", "Fecha Cierre", "Seguimientos (JSON)"
    ]);
  }
  return ss.getActiveSheet();
}
